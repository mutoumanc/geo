地理信息数据测试文件生成方案

先说明边界：

> 我不能生成真实涉密数据。下面所谓“涉密样本”全部是 **模拟涉密样本**，即虚构名称、虚构坐标、虚构属性、虚构场景，用于测试数据分类分级、权限管控、脱敏识别、格式兼容性等。
> 所有模拟敏感数据都不对应任何真实设施、真实坐标、真实单位或真实业务。

---

# 1. 你这 56 种格式的可生成性分级

不是所有格式都能用开源脚本稳定生成。可以分成三类。

## A 类：可以直接用 Python 生成真实可打开文件

下面这些可以通过 Python 直接生成：

```text
shp
dxf
kml
kmz
jpeg
bmp
png
tif
tiff
hdf
asc
obj
las
laz
ply
pts
off
xyz
stl
```

可选支持：

```text
stp
step
```

`stp/step` 需要安装 `cadquery` 或使用 FreeCAD。

---

## B 类：建议用 GDAL/OGR 从 A 类文件转换生成

这些格式可以从 `shp` 或 `tif` 转换得到：

```text
gdb
tab
mif
dgn
bil
bip
bsq
img
grd
dem
```

其中：

- `gdb`：FileGDB，取决于 GDAL 是否支持写 OpenFileGDB
- `dgn`：GDAL 可写 DGN v7，复杂 DGN 建议用 Bentley MicroStation
- `bil/bip/bsq/img/grd/dem`：建议用 `gdal_translate`
- `tab/mif`：建议用 `ogr2ogr`

---

## C 类：强依赖商业软件或厂商 SDK

以下格式开源环境一般无法稳定“创建真正可打开的原生文件”，建议用对应软件导出：

```text
mdb
dwg
edb
accdb
e00
vot
udd
udbx
sdd
wp
wt
w
adf
pg
iq
ecw
sid
dsm
pod
osgb
osg
osgt
3mx
b3dm
neu
```

注意：

- `dsm` 通常不是固定文件格式，而是数据类型，常用 `dsm.tif` 表示。
- `b3dm` 可以用 Cesium 官方样例或 3D Tiles 工具链生成。
- `osg/osgb/osgt` 可以用 OpenSceneGraph 的 `osgconv` 转换。
- `dwg` 建议用 AutoCAD 或 ODA File Converter。
- `mdb/accdb` 建议用 Access/ArcGIS。
- `udbx/udd/sdd` 建议用 SuperMap。
- `wp/wt/w` 建议用 MapGIS。
- `pod/3mx` 建议用 Bentley Pointools / ContextCapture。
- `ecw/sid` 需要商业 SDK。

# 4. Python 一键生成真实可打开样本

保存为：

```bash
generate_real_openable_samples.py
```

代码如下：

```python
import os
import zipfile
import json
import math
import shutil
from pathlib import Path

import numpy as np
import pandas as pd

from shapely.geometry import Point, LineString, Polygon
import geopandas as gpd

from PIL import Image

import rasterio
from rasterio.transform import from_origin

import h5py
import ezdxf
import laspy
import trimesh


# =========================================================
# 基础配置
# =========================================================

OUT = Path("real_format_samples")
OUT.mkdir(exist_ok=True)

CRS = "EPSG:4326"

public_samples = [
    ("01", "city_poi", "城市POI点位", "point"),
    ("02", "road_centerline", "道路中心线", "line"),
    ("03", "admin_boundary", "行政区边界", "polygon"),
    ("04", "park_greenland", "公园绿地", "polygon"),
    ("05", "bus_route", "公交线路", "line"),
    ("06", "river_network", "河流水系", "line"),
    ("07", "building_footprint", "建筑轮廓", "polygon"),
    ("08", "landuse_patch", "土地利用斑块", "polygon"),
    ("09", "tour_route", "旅游路线", "line"),
    ("10", "sampling_point", "公开采样点", "point"),
]

sensitive_samples = [
    ("01", "facility_boundary", "虚构受控设施边界", "polygon"),
    ("02", "underground_pipeline", "虚构地下管线", "line"),
    ("03", "patrol_route", "虚构巡逻路线", "line"),
    ("04", "key_asset_point", "虚构重点资产点位", "point"),
    ("05", "command_site", "虚构指挥点", "point"),
    ("06", "communication_node", "虚构通信节点", "point"),
    ("07", "surveillance_area", "虚构监控覆盖区", "polygon"),
    ("08", "access_control_zone", "虚构访问控制区", "polygon"),
    ("09", "storage_area", "虚构仓储区", "polygon"),
    ("10", "project_redline", "虚构项目红线", "polygon"),
]

records = []


def ensure_dir(*parts):
    p = OUT.joinpath(*parts)
    p.mkdir(parents=True, exist_ok=True)
    return p


def add_record(group, fmt, sample_id, slug, name_cn, path):
    records.append({
        "group": group,
        "format": fmt,
        "sample_id": sample_id,
        "slug": slug,
        "name_cn": name_cn,
        "path": str(path).replace("\\", "/"),
        "is_real_sensitive": False,
        "note": "sensitive_simulated 表示虚构模拟敏感数据，不是真实涉密数据"
    })


# =========================================================
# 生成矢量几何
# =========================================================

def make_vector_gdf(sample_id, slug, name_cn, geom_type, group):
    """
    使用虚构坐标。为避免与真实设施对应，坐标在小范围合成区域内。
    """
    idx = int(sample_id)
    base_x = 100.0 + idx * 0.1
    base_y = 20.0 + idx * 0.1

    if geom_type == "point":
        geoms = [
            Point(base_x, base_y),
            Point(base_x + 0.01, base_y + 0.02),
            Point(base_x + 0.02, base_y + 0.01),
        ]
    elif geom_type == "line":
        geoms = [
            LineString([
                (base_x, base_y),
                (base_x + 0.02, base_y + 0.01),
                (base_x + 0.04, base_y + 0.03),
            ]),
            LineString([
                (base_x + 0.01, base_y - 0.01),
                (base_x + 0.03, base_y + 0.00),
                (base_x + 0.05, base_y + 0.02),
            ]),
        ]
    else:
        geoms = [
            Polygon([
                (base_x, base_y),
                (base_x + 0.04, base_y),
                (base_x + 0.04, base_y + 0.03),
                (base_x, base_y + 0.03),
                (base_x, base_y),
            ]),
            Polygon([
                (base_x + 0.05, base_y + 0.02),
                (base_x + 0.08, base_y + 0.02),
                (base_x + 0.08, base_y + 0.05),
                (base_x + 0.05, base_y + 0.05),
                (base_x + 0.05, base_y + 0.02),
            ]),
        ]

    df = pd.DataFrame({
        "id": range(1, len(geoms) + 1),
        "sample_id": [sample_id] * len(geoms),
        "slug": [slug] * len(geoms),
        "name_cn": [name_cn] * len(geoms),
        "class": [group] * len(geoms),
    })

    return gpd.GeoDataFrame(df, geometry=geoms, crs=CRS)


# =========================================================
# 1. SHP
# =========================================================

def create_shp(group, samples):
    fmt = "shp"
    out_dir = ensure_dir(group, fmt)

    for sample_id, slug, name_cn, geom_type in samples:
        gdf = make_vector_gdf(sample_id, slug, name_cn, geom_type, group)
        sample_dir = out_dir / f"{fmt}_{group}_{sample_id}_{slug}"
        sample_dir.mkdir(exist_ok=True)
        path = sample_dir / f"{fmt}_{group}_{sample_id}_{slug}.shp"
        gdf.to_file(path, driver="ESRI Shapefile", encoding="utf-8")
        add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 2. KML / KMZ
# =========================================================

def create_kml_kmz(group, samples):
    for fmt in ["kml", "kmz"]:
        out_dir = ensure_dir(group, fmt)

        for sample_id, slug, name_cn, geom_type in samples:
            gdf = make_vector_gdf(sample_id, slug, name_cn, geom_type, group)
            kml_path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.kml"

            # 简单手写 KML，Point/Line/Polygon 都支持
            content = ['<?xml version="1.0" encoding="UTF-8"?>']
            content.append('<kml xmlns="http://www.opengis.net/kml/2.2">')
            content.append("<Document>")
            content.append(f"<name>{name_cn}</name>")

            for _, row in gdf.iterrows():
                geom = row.geometry
                content.append("<Placemark>")
                content.append(f"<name>{row['name_cn']}</name>")
                content.append(f"<description>{group}; {slug}</description>")

                if geom.geom_type == "Point":
                    x, y = geom.x, geom.y
                    content.append("<Point>")
                    content.append(f"<coordinates>{x},{y},0</coordinates>")
                    content.append("</Point>")

                elif geom.geom_type == "LineString":
                    coords = " ".join([f"{x},{y},0" for x, y in geom.coords])
                    content.append("<LineString>")
                    content.append(f"<coordinates>{coords}</coordinates>")
                    content.append("</LineString>")

                elif geom.geom_type == "Polygon":
                    coords = " ".join([f"{x},{y},0" for x, y in geom.exterior.coords])
                    content.append("<Polygon><outerBoundaryIs><LinearRing>")
                    content.append(f"<coordinates>{coords}</coordinates>")
                    content.append("</LinearRing></outerBoundaryIs></Polygon>")

                content.append("</Placemark>")

            content.append("</Document>")
            content.append("</kml>")

            kml_text = "\n".join(content)

            if fmt == "kml":
                with open(kml_path, "w", encoding="utf-8") as f:
                    f.write(kml_text)
                add_record(group, fmt, sample_id, slug, name_cn, kml_path)

            else:
                kmz_path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.kmz"
                temp_doc = out_dir / "doc.kml"
                with open(temp_doc, "w", encoding="utf-8") as f:
                    f.write(kml_text)

                with zipfile.ZipFile(kmz_path, "w", zipfile.ZIP_DEFLATED) as z:
                    z.write(temp_doc, "doc.kml")

                temp_doc.unlink()
                add_record(group, fmt, sample_id, slug, name_cn, kmz_path)

# 3. DXF
# =========================================================

def create_dxf(group, samples):
    fmt = "dxf"
    out_dir = ensure_dir(group, fmt)

    for sample_id, slug, name_cn, geom_type in samples:
        idx = int(sample_id)
        doc = ezdxf.new("R2010")
        msp = doc.modelspace()

        base_x = idx * 10
        base_y = idx * 5

        doc.layers.add(name="SAMPLE_LAYER")

        if geom_type == "point":
            for i in range(3):
                msp.add_point((base_x + i, base_y + i), dxfattribs={"layer": "SAMPLE_LAYER"})
                msp.add_text(
                    f"{name_cn}_{i+1}",
                    dxfattribs={"height": 0.5, "layer": "SAMPLE_LAYER"}
                ).set_placement((base_x + i + 0.2, base_y + i + 0.2))

        elif geom_type == "line":
            msp.add_line((base_x, base_y), (base_x + 10, base_y + 5), dxfattribs={"layer": "SAMPLE_LAYER"})
            msp.add_lwpolyline(
                [(base_x, base_y), (base_x + 3, base_y + 2), (base_x + 8, base_y + 1)],
                dxfattribs={"layer": "SAMPLE_LAYER"}
            )

        else:
            msp.add_lwpolyline(
                [
                    (base_x, base_y),
                    (base_x + 8, base_y),
                    (base_x + 8, base_y + 6),
                    (base_x, base_y + 6),
                    (base_x, base_y),
                ],
                close=True,
                dxfattribs={"layer": "SAMPLE_LAYER"}
            )

        path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.dxf"
        doc.saveas(path)
        add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 4. 栅格：TIF/TIFF/JPEG/PNG/BMP/ASC/HDF
# =========================================================

def make_raster_array(sample_id, group):
    idx = int(sample_id)
    rows, cols = 64, 64
    x = np.linspace(0, 1, cols)
    y = np.linspace(0, 1, rows)
    xx, yy = np.meshgrid(x, y)

    base = np.sin((xx + idx * 0.1) * math.pi * 4) + np.cos((yy + idx * 0.1) * math.pi * 4)
    base = base + idx

    if group == "sensitive_simulated":
        # 模拟敏感样本增加一个虚构高值斑块
        cx, cy = 0.5, 0.5
        blob = np.exp(-((xx - cx) ** 2 + (yy - cy) ** 2) / 0.02) * 20
        base = base + blob

    arr = base.astype(np.float32)
    return arr


def create_rasters(group, samples):
    raster_formats = ["tif", "tiff", "jpeg", "png", "bmp", "asc", "hdf"]

    for fmt in raster_formats:
        out_dir = ensure_dir(group, fmt)

        for sample_id, slug, name_cn, geom_type in samples:
            arr = make_raster_array(sample_id, group)

            transform = from_origin(100.0, 30.0, 0.001, 0.001)

            if fmt in ["tif", "tiff"]:
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.{fmt}"
                with rasterio.open(
                    path,
                    "w",
                    driver="GTiff",
                    height=arr.shape[0],
                    width=arr.shape[1],
                    count=1,
                    dtype=arr.dtype,
                    crs=CRS,
                    transform=transform,
                    nodata=-9999
                ) as dst:
                    dst.write(arr, 1)

            elif fmt in ["jpeg", "png", "bmp"]:
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.{fmt if fmt != 'jpeg' else 'jpg'}"
                img_arr = arr.copy()
                img_arr = img_arr - img_arr.min()
                img_arr = img_arr / img_arr.max()
                img_arr = (img_arr * 255).astype(np.uint8)
                Image.fromarray(img_arr).save(path)

            elif fmt == "asc":
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.asc"
                with open(path, "w", encoding="utf-8") as f:
                    f.write(f"ncols         {arr.shape[1]}\n")
                    f.write(f"nrows         {arr.shape[0]}\n")
                    f.write("xllcorner     100.0\n")
                    f.write("yllcorner     29.936\n")
                    f.write("cellsize      0.001\n")
                    f.write("NODATA_value  -9999\n")
                    for row in arr:
                        f.write(" ".join([f"{v:.3f}" for v in row]) + "\n")

            elif fmt == "hdf":
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.hdf"
                with h5py.File(path, "w") as h:
                    h.create_dataset("value", data=arr)
                    h.attrs["crs"] = CRS
                    h.attrs["group"] = group
                    h.attrs["sample_id"] = sample_id
                    h.attrs["sample_name_cn"] = name_cn
                    h.attrs["note"] = "synthetic test dataset"

            add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 5. 点云：LAS/LAZ/XYZ/PTS
# =========================================================

def create_pointclouds(group, samples):
    formats = ["las", "laz", "xyz", "pts"]

    for fmt in formats:
        out_dir = ensure_dir(group, fmt)

        for sample_id, slug, name_cn, geom_type in samples:
            idx = int(sample_id)
            n = 500

            rng = np.random.default_rng(seed=idx + (1000 if group == "sensitive_simulated" else 0))

            x = idx * 10 + rng.random(n) * 10
            y = idx * 5 + rng.random(n) * 10
            z = rng.random(n) * 5

            if group == "sensitive_simulated":
                # 加一个虚构结构轮廓
                z += np.exp(-((x - (idx * 10 + 5)) ** 2 + (y - (idx * 5 + 5)) ** 2) / 8) * 10

            if fmt in ["las", "laz"]:
                header = laspy.LasHeader(point_format=3, version="1.2")
                header.offsets = np.array([0.0, 0.0, 0.0])
                header.scales = np.array([0.001, 0.001, 0.001])

                las = laspy.LasData(header)
                las.x = x
                las.y = y
                las.z = z
                las.intensity = rng.integers(0, 255, n, dtype=np.uint16)

                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.{fmt}"
                las.write(path)

            elif fmt == "xyz":
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.xyz"
                data = np.column_stack([x, y, z])
                np.savetxt(path, data, fmt="%.4f")

            elif fmt == "pts":
                path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.pts"
                data = np.column_stack([x, y, z, np.ones(n) * 128])
                with open(path, "w", encoding="utf-8") as f:
                    f.write(f"{n}\n")
                    np.savetxt(f, data, fmt="%.4f %.4f %.4f %.0f")

            add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 6. 三维模型：OBJ/PLY/STL/OFF
# =========================================================

def make_mesh(sample_id, group):
    idx = int(sample_id)

    if group == "public":
        mesh = trimesh.creation.box(extents=(1 + idx * 0.05, 1, 1))
    else:
        # 虚构敏感：组合几个盒体，模拟建筑/设施，但不对应真实结构
        base = trimesh.creation.box(extents=(1.5, 1.0, 0.5))
        tower = trimesh.creation.box(extents=(0.4, 0.4, 1.5))
        tower.apply_translation((0.3, 0.2, 1.0))
        mesh = trimesh.util.concatenate([base, tower])

    mesh.apply_translation((idx * 2, 0, 0))
    return mesh


def create_meshes(group, samples):
    formats = ["obj", "ply", "stl", "off"]

    for fmt in formats:
        out_dir = ensure_dir(group, fmt)

        for sample_id, slug, name_cn, geom_type in samples:
            mesh = make_mesh(sample_id, group)
            path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.{fmt}"
            mesh.export(path)
            add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 7. 可选：STEP/STP
# =========================================================

def create_step_optional(group, samples):
    """
    如果安装了 cadquery，则生成真正的 STEP/STP。
    安装方式：
        conda install -c conda-forge cadquery -y
    """
    try:
        import cadquery as cq
    except Exception:
        print("未安装 cadquery，跳过 stp/step。")
        return

    for fmt in ["stp", "step"]:
        out_dir = ensure_dir(group, fmt)

        for sample_id, slug, name_cn, geom_type in samples:
            idx = int(sample_id)

            if group == "public":
                model = cq.Workplane("XY").box(10 + idx, 10, 5)
            else:
                model = (
                    cq.Workplane("XY")
                    .box(12, 8, 4)
                    .faces(">Z")
                    .workplane()
                    .box(4, 4, 8)
                )

            path = out_dir / f"{fmt}_{group}_{sample_id}_{slug}.{fmt}"
            cq.exporters.export(model, str(path))
            add_record(group, fmt, sample_id, slug, name_cn, path)


# =========================================================
# 主流程
# =========================================================

def run():
    tasks = [
        ("public", public_samples),
        ("sensitive_simulated", sensitive_samples),
    ]

    for group, samples in tasks:
        create_shp(group, samples)
        create_kml_kmz(group, samples)
        create_dxf(group, samples)
        create_rasters(group, samples)
        create_pointclouds(group, samples)
        create_meshes(group, samples)
        create_step_optional(group, samples)

    # 输出清单
    manifest = OUT / "manifest.csv"
    metadata = OUT / "metadata.json"

    df = pd.DataFrame(records)
    df.to_csv(manifest, index=False, encoding="utf-8-sig")

    with open(metadata, "w", encoding="utf-8") as f:
        json.dump(records, f, ensure_ascii=False, indent=2)

    print("生成完成：", OUT)
    print("样本数量：", len(records))
    print("清单：", manifest)
    print("元数据：", metadata)


if __name__ == "__main__":
    run()
```

代码如下：

```python
import os
import subprocess
from pathlib import Path

BASE = Path("real_format_samples")

groups = ["public", "sensitive_simulated"]

def run_cmd(cmd):
    print("运行：", " ".join(str(c) for c in cmd))
    try:
        subprocess.run(cmd, check=True)
    except Exception as e:
        print("失败：", e)

def find_first_file(folder, suffix):
    files = list(folder.rglob(f"*.{suffix}"))
    return files

for group in groups:
    # -----------------------------------------------------
    # 矢量转换：shp -> gdb/tab/mif/dgn
    # -----------------------------------------------------
    shp_root = BASE / group / "shp"

    for shp in shp_root.rglob("*.shp"):
        stem = shp.stem

        # FileGDB
        out_dir = BASE / group / "gdb"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_gdb = out_dir / f"{stem}.gdb"
        if not out_gdb.exists():
            run_cmd([
                "ogr2ogr",
                "-f", "OpenFileGDB",
                str(out_gdb),
                str(shp)
            ])

        # MapInfo TAB
        out_dir = BASE / group / "tab"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_tab = out_dir / f"{stem}.tab"
        run_cmd([
            "ogr2ogr",
            "-f", "MapInfo File",
            str(out_tab),
            str(shp)
        ])

        # MapInfo MIF
        out_dir = BASE / group / "mif"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_mif = out_dir / f"{stem}.mif"
        run_cmd([
            "ogr2ogr",
            "-f", "MapInfo File",
            str(out_mif),
            str(shp)
        ])

        # DGN
        out_dir = BASE / group / "dgn"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_dgn = out_dir / f"{stem}.dgn"
        run_cmd([
            "ogr2ogr",
            "-f", "DGN",
            str(out_dgn),
            str(shp)
        ])

    # -----------------------------------------------------
    # 栅格转换：tif -> bil/bip/bsq/img/grd/dem
    # -----------------------------------------------------
    tif_root = BASE / group / "tif"

    for tif in tif_root.rglob("*.tif"):
        stem = tif.stem

        # BIL，EHdr
        out_dir = BASE / group / "bil"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_bil = out_dir / f"{stem}.bil"
        run_cmd([
            "gdal_translate",
            "-of", "EHdr",
            str(tif),
            str(out_bil)
        ])

        # BIP，ENVI
        out_dir = BASE / group / "bip"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_bip = out_dir / f"{stem}.bip"
        run_cmd([
            "gdal_translate",
            "-of", "ENVI",
            "-co", "INTERLEAVE=BIP",
            str(tif),
            str(out_bip)
        ])

        # BSQ，ENVI
        out_dir = BASE / group / "bsq"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_bsq = out_dir / f"{stem}.bsq"
        run_cmd([
            "gdal_translate",
            "-of", "ENVI",
            "-co", "INTERLEAVE=BSQ",
            str(tif),
            str(out_bsq)
        ])

        # ERDAS IMG
        out_dir = BASE / group / "img"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_img = out_dir / f"{stem}.img"
        run_cmd([
            "gdal_translate",
            "-of", "HFA",
            str(tif),
            str(out_img)
        ])

        # Surfer ASCII Grid
        out_dir = BASE / group / "grd"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_grd = out_dir / f"{stem}.grd"
        run_cmd([
            "gdal_translate",
            "-of", "GSAG",
            str(tif),
            str(out_grd)
        ])

        # USGS DEM
        out_dir = BASE / group / "dem"
        out_dir.mkdir(parents=True, exist_ok=True)
        out_dem = out_dir / f"{stem}.dem"
        run_cmd([
            "gdal_translate",
            "-of", "USGSDEM",
            str(tif),
            str(out_dem)
        ])

print("GDAL 转换完成。")
```

# 9. 生成 DWG 的推荐流程

开源脚本通常不能稳定写 DWG。建议：

## 9.1 先生成 DXF

上面的脚本已经生成：

```text

real_format_samples/sensitive_simulated/dxf/
```

## 9.2 用 ODA File Converter 转 DWG

下载：

```text
https://www.opendesign.com/guestfiles/oda_file_converter
```

使用 GUI 或命令行将 DXF 批量转为 DWG。

```bash
ODAFileConverter input_dxf_dir output_dwg_dir ACAD2018 DWG 0 1 "*.dxf"
```
